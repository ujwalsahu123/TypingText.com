// Function to clear the screen
void clearScreen() {
#ifdef _WIN32
    system("cls");
#else
    system("clear");
#endif
}

// Function to set non-blocking input for Unix-based systems
void setNonBlocking(bool enable) {
#ifndef _WIN32
    struct termios ttystate;
    tcgetattr(STDIN_FILENO, &ttystate);
    if (enable) {
        ttystate.c_lflag &= ~(ICANON | ECHO);
        ttystate.c_cc[VMIN] = 0;
    } else {
        ttystate.c_lflag |= ICANON | ECHO;
    }
    tcsetattr(STDIN_FILENO, TCSANOW, &ttystate);
#endif
}

int main() {
    std::vector<std::string> elements;
    std::string input;

    int numLetters;
    std::cout << "Enter the number of letters in each element: ";
    std::cin >> numLetters;

    std::cout << "Enter elements separated by spaces and press enter to start the game: ";
    std::getline(std::cin, input); // Consume the newline
    std::getline(std::cin, input);
    std::istringstream iss(input);
    std::string token;
    while (iss >> token) {
        if (token.size() == numLetters)
            elements.push_back(token);
        else
            std::cerr << "Ignoring element '" << token << "' as it does not have " << numLetters << " letters." << std::endl;
    }

    // Seed for random number generator
    std::srand(std::time(nullptr));

    char exitKey;
    do {
        // Shuffle the elements randomly before each iteration
        std::random_shuffle(elements.begin(), elements.end());

        clearScreen();
        for (const auto& element : elements) {
            clearScreen();
            std::cout << element << std::endl;
            std::string userGuess;

#ifdef _WIN32
            while (!_kbhit()); // Wait for keypress on Windows systems
            userGuess = _getch();
#else
            setNonBlocking(true); // Set non-blocking input for Unix-based systems
            for (int i = 0; i < numLetters; ++i) {
                char c;
                while (read(STDIN_FILENO, &c, 1) != 1); // Wait for keypress
                userGuess += c;
            }
            setNonBlocking(false); // Restore blocking input
#endif

            if (userGuess == "1")
                return 0; // Exit the program if user inputs '1'

            while (userGuess != element) {
                std::cout << "Wrong answer, type again" << std::endl;
                userGuess = ""; // Reset userGuess to empty

#ifdef _WIN32
                while (!_kbhit()); // Wait for keypress on Windows systems

#else
